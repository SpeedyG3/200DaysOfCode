class Solution {
    public int totalMoney(int n) {
        int ans = 0;
        int w = n / 7;

        // Calculate the total for full weeks
        for (int i = 0; i < w; i++) {
            ans = ans + (28 + (i * 7));
        }

        int d = n % 7;

        // Calculate the total for remaining days
        for (int i = 1; i <= d; i++) {
            ans = ans + (w + i);
        }

        return ans;
    }
}
