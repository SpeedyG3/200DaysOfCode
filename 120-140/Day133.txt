1) 1. Two Sum
class Solution {
    public int[] twoSum(int[] nums, int target) {
        int n = nums.length;
        HashMap<Integer, Integer> hm = new HashMap<>();
        int[] result = new int[2];
        for(int i=0;i<n;i++){
            if(hm.containsKey(target-nums[i])){
                result[0] = hm.get(target-nums[i]);
                result[1] = i;
                return result;
            }
            hm.put(nums[i], i);
        }
        return result;
    }
}
